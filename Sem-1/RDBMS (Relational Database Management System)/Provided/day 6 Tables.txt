
CREATE TABLE Worker (
    Id INT PRIMARY KEY,
    Name VARCHAR(12),
    Wages_Per_Hr DECIMAL(10,2)
);

CREATE TABLE Job (
    Job_Id INT PRIMARY KEY,
    Job_Type VARCHAR(15)
);

CREATE TABLE Assigned (
    W_Id INT,
    J_Id INT,
    Start_Date DATE,
    Status VARCHAR(50),
    Total_Hrs INT,
    FOREIGN KEY (W_Id) REFERENCES Worker(Id),
    FOREIGN KEY (J_Id) REFERENCES Job(Job_Id)
);
