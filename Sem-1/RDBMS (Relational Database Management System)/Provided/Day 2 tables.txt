CREATE TABLE Sales_Order (
    Order_No     VARCHAR2(4) PRIMARY KEY CHECK (Order_No LIKE 'O%'),
    Client_No    VARCHAR2(5),
    Order_Date   DATE,
    Salesman_No  VARCHAR2(4) REFERENCES Salesman_Master(S_No),
    Delivery_Type CHAR(1) CHECK (Delivery_Type IN ('F','P')),
    Bill_Paid    CHAR(1) CHECK (Bill_Paid IN ('Y','N')),
    Delivery_Date DATE,
    Order_Status VARCHAR2(20) CHECK (Order_Status IN ('In Process','Pending','Fulfilled'))
);

CREATE TABLE Sales_Order_Details (
    Order_No       VARCHAR2(4),
    Product_No     VARCHAR2(4),
    Qty_Ordered    NUMBER(6) CHECK (Qty_Ordered > 0),
    Qty_Dispatched NUMBER(6) CHECK (Qty_Dispatched >= 0),
    Product_Rate   NUMBER(10,2) CHECK (Product_Rate >= 0),
    PRIMARY KEY (Order_No, Product_No),
    FOREIGN KEY (Order_No) REFERENCES Sales_Order(Order_No),
    FOREIGN KEY (Product_No) REFERENCES Product_Master(P_No)
);

/* =========================================
   4. APPLICANT / ENTRANCE TEST / ETEST_DETAILS
   ========================================= */
CREATE TABLE Applicant (
    AID     VARCHAR2(4) PRIMARY KEY CHECK (AID LIKE 'A%'),
    A_Name  VARCHAR2(30) NOT NULL,
    Address VARCHAR2(50),
    B_Date  DATE
);

CREATE TABLE Entrance_Test (
    ET_ID     VARCHAR2(4) PRIMARY KEY CHECK (ET_ID LIKE 'T%'),
    ET_Name   VARCHAR2(30) NOT NULL,
    Max_Score NUMBER(3) CHECK (Max_Score <= 100)
);

CREATE TABLE ETest_Details (
    AID        VARCHAR2(4),
    ETID       VARCHAR2(4),
    ETest_Date DATE,
    Score      NUMBER(3) CHECK (Score >= 0),
    PRIMARY KEY (AID, ETID),
    FOREIGN KEY (AID) REFERENCES Applicant(AID),
    FOREIGN KEY (ETID) REFERENCES Entrance_Test(ET_ID)
);

/* =========================================
   5. DISTRIBUTOR / ITEM / DIST_ITEM
   ========================================= */
CREATE TABLE Distributor (
    Dno    VARCHAR2(4) PRIMARY KEY,
    DName  VARCHAR2(30) NOT NULL,
    City   VARCHAR2(30),
    Phone  VARCHAR2(12)
);

CREATE TABLE Item (
    Item_No   VARCHAR2(5) PRIMARY KEY CHECK (Item_No LIKE 'I%'),
    Item_Name VARCHAR2(30) NOT NULL,
    Price     NUMBER(8,2) NOT NULL CHECK (Price > 0),
    Weight    NUMBER(8,2) NOT NULL CHECK (Weight > 0)
);

CREATE TABLE Dist_Item (
    Dno     VARCHAR2(4),
    Item_No VARCHAR2(5),
    Qty     NUMBER(6) NOT NULL CHECK (Qty > 0),
    Order_Date DATE,
    PRIMARY KEY (Dno, Item_No),
    FOREIGN KEY (Dno) REFERENCES Distributor(Dno),
    FOREIGN KEY (Item_No) REFERENCES Item(Item_No)
);
