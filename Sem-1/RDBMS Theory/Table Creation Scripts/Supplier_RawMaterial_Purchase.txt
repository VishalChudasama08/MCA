S_ID varchar(4) check(S_ID like 'S%') primary key,
S_NAME varchar(15) not null,
contact number(10) check(length(contact)=10) not null
);

S_ID varchar(4) not null,

constraint S_ID foreign key(S_ID) references SUPPLIER(S_ID),
constraint id foreign key(id) references RAWMATIRIAL(id)
