create table SUPPLIER(
s_id varchar(5) primary key,
s_name varchar(30) not null,
contact number(10) check(length(contact)=10) not null,
Contact_Person VARCHAR(20) NOT NULL);


 create table PARTS(
part_id varchar(5) primary key,
 part_name varchar(50) not null,
 unit_rate number(20) not null);


create table CATALOG(
s_id varchar(5),
part_id varchar(5),
qty number(10) not null,
 constraint s_id foreign key(s_id) references SUPPLIER on delete cascade,
 	constraint part_id foreign key(part_id) references PARTS on delete set NULL);
