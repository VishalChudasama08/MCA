create table EMPLOYEE(
empid varchar(5) check(empid like 'E%') primary key,
ename varchar(10) not null,
J_date date not null,
designation varchar(10) not null
);

Table created.


create table PROJECT(
pid varchar(5) check(pid like 'P%') primary key,
pname varchar(30) not null,
proj_type varchar(30) not null,
tot_duration int not null check (tot_duration BETWEEN 7 AND 30)
);

Table created.

create table WORKS_ON(
empid varchar(5) check(empid like 'E%'),
pid varchar(5) check(pid like 'P%'),
type_work varchar(10) not null,
num_days int not null,
constraint fk_empid foreign key (empid) references EMPLOYEE(empid),
constraint fk_pid foreign key (pid) references PROJECT(pid)
);
